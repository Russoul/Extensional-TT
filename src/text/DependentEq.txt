
(B : A → 𝕌) (e : a₀ ≡ a₁ ∈ A) (b₀ : B a₀) (b₁ : B a₁) ⊦
  b₀ ≡ b₁ ∈ B / e : 𝕌
  (b₀ ≡ b₁ ∈ B / e) ≔ transp B e b₀ ≡ b₁ ∈ B a₁


(B : A → 𝕌) (b₀ b₁ : B a) ⊦
  _ : (b₀ ≡ b₁ ∈ B / Refl) ≡ (b₀ ≡ b₁ ∈ B a) ∈ 𝕌
  _ : (transp B Refl b₀ ≡ b₁ ∈ B a) ≡ (b₀ ≡ b₁ ∈ B a) ∈ 𝕌
  Refl : (b₀ ≡ b₁ ∈ B a) ≡ (b₀ ≡ b₁ ∈ B a) ∈ 𝕌

// dependent composition
_ : (l : a₀ ≡ a₁) (p : b₀ ≡ b₁ ∈ B / l) (e : a₁ ≡ a ∈ A) (f : b₁ ≡ b ∈ B / e) → b₀ ≡ b ∈ B / (l ∙ e)
_ Refl (p : b₀ ≡ b₁ ∈ B / Refl) Refl (f : b₁ ≡ b ∈ B / Refl) =
           ? : b₀ ≡ b ∈ B / (Refl ∙ Refl)
           ? : b₀ ≡ b ∈ B / Refl
           ? : b₀ ≡ b ∈ B a
